### 受控组件

export const ControlledSwitch = () => {
  const [selected, setSelected] = useState(false)
  const handleChange = useCallback((currSelected) => {
    setSelected(currSelected)
  }, [])
  return <Switch label="controlled" isSelected={selected} onChange={handleChange} />
}
